eric vinicius do nascimento pereire 1a formacao de docentes // Seleciona o botão de alternar tema const themeToggle = document.getElementById('theme-toggle'); // Verifica se o usuário já tem uma preferência salva no navegador const currentTheme = localStorage.getItem('theme'); if (currentTheme) { document.documentElement.setAttribute('data-theme', currentTheme); } // Adiciona o evento de clique para mudar o tema themeToggle.addEventListener('click', () => { // Verifica o tema atual let theme = document.documentElement.getAttribute('data-theme'); // Alterna entre light e dark if (theme === 'dark') { document.documentElement.setAttribute('data-theme', 'light'); localStorage.setItem('theme', 'light'); } else { document.documentElement.setAttribute('data-theme', 'dark'); localStorage.setItem('theme', 'dark'); } });
